void main(void) {
    cfg_io();
    cfg_clk();

    //*************************************************************************
    // Loop infinito
    //*************************************************************************
    while(1){
        TMR1H = 0;
        TMR1L = 0;
        
        RA0 = 1;
        __delay_us(10);
        RA0 = 0;

        
        while(!RA1);              //Waiting for Echo goes HIGH
        TMR1ON = 1;               //Timer Starts
        while(RA1);               //Waiting for Echo goes LOW
        TMR1ON = 0;               //Timer Stops
        
        V = (TMR1L | (TMR1H<<8));
        D = V/58;
        
        if(D >= 20){
            RB1 = 0;
            i1 = 0;
        }
        if(D < 20){
            RB1 = 1;
            i1 = 1;
        }
        
        if(RA2 == 1){
            RB0 = 1;
            i0 = 1;
        }
        if(RA2 == 0){
            RB0 = 0;
            i0 = 0;
        }
        __delay_ms(500);
    }
    return;
}
//*****************************************************************************
// Funci�n de Inicializaci�n

void cfg_io(void){
    ANSEL = 0;
    ANSELH = 0;
    
    TRISAbits.TRISA0 = 0;
    TRISAbits.TRISA1 = 1;
    TRISAbits.TRISA2 = 1;
    TRISB = 0;
    TRISD = 0;
    TRISB = 0;
    
    PORTB = 0;
    PORTD = 0;
    I2C_Slave_Init(0x50);  
}
void cfg_clk(){
    OSCCONbits.IRCF = 0b111; //IRCF = 110 (4MHz) 
    OSCCONbits.SCS = 1;   //Reloj interno habilitado
}
